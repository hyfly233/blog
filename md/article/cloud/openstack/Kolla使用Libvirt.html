OpenStack 容器中使用 Libvirt

OpenStack 容器中集成了 libvirt，进入容器 nova_libvirt 可以使用 virsh 命令查看虚机列表、虚机详情、虚机资源使用等信息。

# step.1 进入 docker 容器 nova_libvirt
docker exec -it nova_libvirt /bin/sh

# step.2 查看虚机列表
virsh list

Golang 调用 libvirt

https://pkg.go.dev/libvirt.org/go/libvirt#section-readme

# step.1
yum install libvirt libvirt-devel pkgconfig -y

# step.2 安装 golang

Go 示例代码

package main

import (
    "fmt"
    "libvirt.org/go/libvirt"
)

func main() {
    conn, err := libvirt.NewConnect("qemu:///system")

    if err != nil {
        panic(err)
    }
    defer conn.Close()

    doms, err := conn.ListAllDomains(libvirt.CONNECT_LIST_DOMAINS_ACTIVE)
    if err != nil {
        panic(err)
    }

    fmt.Printf("%d running domains:\n", len(doms))
    for _, dom := range doms {
        name, err := dom.GetName()
        if err == nil {
            fmt.Printf("  %s\n", name)
        }
        dom.Free()
    }
}
